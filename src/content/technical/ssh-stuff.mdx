<section className="mt-6">
	<h2 className="text-xl font-bold">SSH</h2>
	<ol className="list-disc p-4 w-[20rem] lg:w-full">
		<li>
			<h3 className="text-md font-semibold">Setting up aliases</h3>
			<p className="mt-2">Create a config file under `~/.ssh` (<i>tab on second line is important</i>)</p>
			<p className="mt-4">
			```bash title="~/.ssh/config"
			Host myHost
		        HostName MyHostName
		        User ec2-user
		        IdentityFile ~/.ssh/identity.pem
			```
			</p>
		</li>
	</ol>
</section>
<section className="mt-6">
	<h2 className="text-xl font-bold">tmux</h2>
	<ol className="list-disc p-4 flex flex-col gap-4 mt-4 w-[20rem] lg:w-full">
		<li>
			<h3 className="text-md font-semibold">Start session</h3>
			<p className="mt-2">
				```bash showLineNumbers=false
					tmux
				```
			</p>
		</li>
		<li>
			<h3 className="text-md font-semibold">Detach from session</h3>
			<p className="mt-4">
				`Ctrl-b` + `D`
			</p>
		</li>
		<li>
			<h3 className="text-md font-semibold">Attach to session</h3>
			<p className="mt-4 w-[20rem] lg:w-full">
				```bash showLineNumbers=false
					tmux ls
					tmux attach -t 0
				```
			</p>
		</li>
		<li>
			<h3 className="text-md font-semibold">Delete session</h3>
			<p className="mt-4 w-[20rem] lg:w-full">
				```bash showLineNumbers=false
					tmux kill-session -t 0
				```
			</p>
		</li>
	</ol>
</section>
